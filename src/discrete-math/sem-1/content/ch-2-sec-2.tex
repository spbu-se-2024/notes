\section{Бинарные отношения}\label{sec:ch-2-sec-2}

\begin{definition}[Свойства бинарных отношений]
    Пусть $A$ --- множество, $R \subseteq A \times A$, тогда:

    \begin{compactitem}
        \item $R$ \definiendum{рефлексивно}, если $\forall a \in A\ (a, a) \in R$.
        \item $R$ \definiendum{симметрично}, если $\forall a, b \in A\ (a, b) \in R \iff (b, a) \in R$.
        \item $R$ \definiendum{транзитивно}, если $\forall a, b, c \in A\ (a, b) \in R \land (b, c) \in R \imp (a, c) \in R$.
        \item $R$ \definiendum{антирефлексивно}, если $\forall a \in A\ (a, a) \notin R$.
        \item $R$ \definiendum{антисимметрично}, если $\forall a \in A\ (a, b) \in R \land (b, a) \in R \imp a = b$.
        \item $R$ \definiendum{антитранзитивно}, если $\forall a, b, c \in A\ (a, b) \in R \land (b, c) \in R \imp (a, c) \notin R$.
        \item $R$ \definiendum{асимметрично}, если $\forall a \in A\ (a, b) \in R \imp (b, a) \notin R$.
    \end{compactitem}
\end{definition}
